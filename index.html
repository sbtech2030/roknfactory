من نحن

ركــن التطــور هــي شــركة شــاملة لعــدة مجــالات بالمقــاولات العامــة تأسســت عــام 1439هــ بالمملكــة العربيـة السـعودية، وتمتلـك ثـلاث مصانـع خاصـة بهـا ذات تقنيـة تكنولوجيـة حديثـة لتصنيـع الكتل الخرسـانية الجاهــزة حيــث يعمــل المصنــع علــى إنتــاج "غــرف التفتيــش، المناهــل الخرســانية مســبقة الصــب، الحواجــز الأمنيـة، وقواعـد أعمـدة إنـارة الشـوارع، بالإضافـة إلـى صناعـة كل مـا يخـص الخرسـانة مسـبقة الصـب.

كمـا إنـه لدينـا مصنـع خـاص بصناعـة الإكسسـوار مـن السـلالم البلاسـتيكية والتـي يتـم صناعتهـا علـى أعلى درجات الجـودة مـن مادتـي الحديـد أوالصلـب او السـتانلس سـتيل المغطـى بطبقـة بلاسـتيكية عازلـة ضـد الصـدأ مـن "البولـي إيثليـن" المطابقـة للمعاييـر العالميـة والمحليـة مـع تخصصنـا فـي صناعـة جميـع أنـواع المعـدات اللازمـة لمعالجـة الميـاه وصناعـة محطـات تحليـة الميـاه بمختلـف الأحجـام والتقنيـات وأيضًـا صناعـة الفلاتـر والشـمعات الخاصـة بـه مـن خيـوط "البولـي بربيليـن، والسـتانلس سـتيل."

بالإضافـة إلـى توفيرنـا الإكسسـوار اللازمـة لتنفيـذ أعمـال البنيـة التحتيـة لمشـاريع "الاتصـالات، الكهربـاء، الصـرف الصحـي، والطـرق" باسـتخدام الحديــد الصناعــي، وبذلــك قمنــا بتنفيــذ العديــد مــن المشــاريع فــي جميــع أنحــاء المملكــة وأيضً ــا مشــاريع محطـات تحليـة ميـاه الابـار والبحـر والصـرف الصحـي، وذلـك مـن خـلال فريقًـا مـن الفنييـن المحترفيـن ذو الخبـرة العاليـة فـي هـذا المجـال.

كمـا إن شـركتنا كانـت مـن ضمن مجموعـة اقتصادية ُمتميـزة داخل المملكة ً لمـا كان لهـا دور ً ا أساسـيا بالسـوق المحلـي مـن خـلال فتـح أبـواب الفـرص الاسـتثمارية المحليـة والأجنبيـة الغنيـة بالإبـداع والأفـكار المتطـورة. ويشـهد علينـا التاريـخ مـن الخبـرة الطويلـة فـي تزويـد الكثيـر مـن المشـاريع بـكل مـا تطلبـه بقـدرة إنتاجيـة وطاقـة تشـغيلية عاليـة بالتقنيـات حديثـة وكـوادر مهنيـة ذو كفـاءات احترافيـة فـي أداء كل عمـل علـى أكمـل وجـه والوصـول إلـى أفضـل النتائـج.

رؤيتنا

نتطلــع إلــى تعزيــز الثقــة بيننــا وبيــن عملائنــا وتكويــن علاقــات طويلــة الأمــد بمنتجــات موثوقــة وتُ لبــي احتياجـات مشـروعهم مـع الاحتفـاظ بأعلـى جـودة ممكنـة، وأن ُنصبـح الشـركة الرائـدة بالمملكـة التـي تقـوم بتنفيــذ جميــع المشــاريع الخاصــة بتحليــة الميــاه والصــرف الصحــي وبالتالــي فتــح بــاب الاســتثمار الأجنبــي بالســوق الســعودي وتحقيــق رؤيــة المملكــة 2030

استراتيجية عملنا

بالإضافـة إلـى التنظيـم والشـفافية التـي يقـوم عليهـم أسـاس عملنـا للتنظيـم الدائـم واتخـاذ القـرارت الصحيحـة فـي اختيـار شـركاؤنا ومنتجاتنـا بعنايـة فائقـة، وبالتالـي تحسـين الرقابـة ورفـع المسـتوى لـدى جميـع العامليـن ً لدينـا بأهميـة الـدور الـذي نقـوم بـه تجـاه تحقيـق أهـداف الشـركة والنهـوض بهـا للأمـام دائمـا ورفـع الجـودة وقيمـة المنتجـات بمـا يليـق بمجتمعنـا، ولتطبيـق هـذا العمـل بشـكل متكامـل فإننـا نبنـي اسـتراتيجية العمـل ُ فـي سـياقات متميـزة تتوافـق مـع رؤيتنـا ومهمتنـا والدعـم الصحيـح لكافـة أنشـطة الشـركة.
